    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
        if ((Validaciones.validarNumeros(puntajeEmpate.getText()))
                && (Validaciones.validarNumeros(puntajeGanador.getText()))
                && (Validaciones.validarNumeros(puntajeResExacto.getText()))
                && (Validaciones.letras(nombrePenca.getText()))) {
            int aciertoGanador = Integer.parseInt(puntajeGanador.getText());
            int puntajeEmpate = Integer.parseInt(this.puntajeEmpate.getText());
            int goles = Integer.parseInt(puntajeResExacto.getText());
            String nombrePenc = nombrePenca.getText();
            if ((Validaciones.validarMaximoPuntaje(goles))
                    && (Validaciones.validarMaximoPuntaje(aciertoGanador))
                    && (Validaciones.validarMaximoPuntaje(puntajeEmpate))) {
                Reglas reglas = new Reglas(aciertoGanador, goles, puntajeEmpate);
                if(this.sep.isSelected()){
                    reglas.setAumentoEnCambioDeFase(true);
                }
                Juego penca = new Juego();
                penca.setReglas(reglas);
                penca.agregarParticipante(getSistema().getJugadorQueUsa());
                penca.setNombreDePenca(nombrePenc);
                getSistema().guardarSistema(getSistema());
                getSistema().setJuego(penca);
                JOptionPane.showMessageDialog(this,"Penca creada con exito","Se creo la penca", 1);
                menu.getCrearPenca().setEnabled(false);
                getSistema().setExistePenca(true);
                getSistema().guardarSistema(getSistema());
            } else {
                JOptionPane.showMessageDialog(this, "Los puntajes varian de 1-10 y el nombre de la penca solo puede tener letras, porfavor re ingrese.", "Atencion", 2);
            }

        } else {
            JOptionPane.showMessageDialog(this, "Ingrese los campos de manera correcta", "Atencion", 2);
        }


    }//GEN-LAST:event_jButton1ActionPerformed
