    private void rankingsMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_rankingsMouseClicked
        String rank = (String) rankings.getSelectedValue();

        if (!rank.isEmpty()) {
            String[] datos = rank.split("-");
            String alias = datos[1];
            int points = this.getSistema().buscarPuntaje(alias);
            this.puntaje.setText("Puntaje : "+points);
        }else{
            JOptionPane.showMessageDialog(this,"Debe seleccionar un jugador de la lista para ver el puntaje","Selecione un jugador", 2);
        }
    }//GEN-LAST:event_rankingsMouseClicked
