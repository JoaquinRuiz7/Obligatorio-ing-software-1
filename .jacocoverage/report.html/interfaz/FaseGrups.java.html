    private void gaMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_gaMouseClicked
        Partido partido = (Partido) ga.getSelectedValue();
        if (partido != null) {
            this.setPartido(partido);
            String nombrePais1 = partido.getEquipos()[0];
            String nombrePais2 = partido.getEquipos()[1];
            this.pais1.setText(nombrePais1);
            this.pais2.setText(nombrePais2);
        }

    }//GEN-LAST:event_gaMouseClicked

    private void pronosticarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_pronosticarActionPerformed
        String resultado1 = this.resultado1.getText();
        String resultado2 = this.resultado2.getText();
        if (partido != null && (!resultado1.isEmpty()) && (!resultado2.isEmpty())) {
            int resultadoP1 = Integer.parseInt(this.resultado1.getText());
            int resultadoP2 = Integer.parseInt(this.resultado2.getText());
            int[] resultado = new int[2];
            resultado[0] = resultadoP1;
            resultado[1] = resultadoP2;
            PronosticoPartido pronostico = new PronosticoPartido(resultado, getPartido());
            if (!(getSistema().getJugadorQueUsa().getPronosticos().contains(pronostico))) {
                this.getSistema().getJugadorQueUsa().getPronosticos().add(pronostico);
                this.getSistema().guardarSistema(getSistema());
                pronosticos.setListData(getSistema().getJugadorQueUsa().getPronosticos().toArray());
                JOptionPane.showMessageDialog(this, "Tu pronostico para el partido " + partido.getEquipos()[0] + " VS " + partido.getEquipos()[1] + " fue registrado con exito.", "Pronostico registrado", 1);
            } else {
                JOptionPane.showMessageDialog(this, "Ya se pronostico ese partido", "Ya se hzo el pronostico", 1);
            }

        } else {
            JOptionPane.showMessageDialog(this, "Porfavor seleccione el partido a apostar y ingrese el resultado a prnosticar.", "Atencion , llene los campos", 2);
        }
    }//GEN-LAST:event_pronosticarActionPerformed
