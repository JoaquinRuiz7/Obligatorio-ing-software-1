    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
          Partido final1 = partidos.get(0);
          Partido final2 = partidos.get(1);
        String resultado1 = this.r1.getText();
        String resultado2 = this.r2.getText();
        String resultado3 = this.r3.getText();
        String resultado4 = this.r4.getText();
        if ((partido != null) && (!(resultado1.isEmpty())) && (!(resultado2.isEmpty()))
                && (!resultado3.isEmpty()) && (!resultado4.isEmpty())) {
            int resultadoP1 = Integer.parseInt(this.r1.getText());
            int resultadoP2 = Integer.parseInt(this.r2.getText());
            int[] resultado = new int[2];
            resultado[0] = resultadoP1;
            resultado[1] = resultadoP2;
            PronosticoPartido pronostico = new PronosticoPartido(resultado, final1);
             PronosticoPartido pronostico2 = new PronosticoPartido(resultado, final2);
            if (!(getSistema().getJugadorQueUsa().getPronosticos().contains(pronostico))) {
                this.getSistema().getJugadorQueUsa().getPronosticos().add(pronostico);
                this.getSistema().getJugadorQueUsa().getPronosticos().add(pronostico2);
                this.pronostico.setListData(getSistema().getJugadorQueUsa().getPronosticos().toArray());
                JOptionPane.showMessageDialog(this, "Tu pronostico fue registrado cone exito" , "Pronostico registrado", 1);
            } else {
                JOptionPane.showMessageDialog(this, "Ya se hizo el pronostico", "Ya se pronostico ese partido", 1);
            }
        } else {
            JOptionPane.showMessageDialog(this, "Porfavor seleccione el partido a apostar y ingrese el resultado a prnosticar.", "Atencion , llene los campos", 2);
        }
    }//GEN-LAST:event_jButton1ActionPerformed
