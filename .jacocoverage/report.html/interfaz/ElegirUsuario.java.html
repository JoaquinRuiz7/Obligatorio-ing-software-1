    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
     Jugador seleccionado = (Jugador) this.seleccionarJugador.getSelectedValue();
     if(seleccionado!=null){
         sistema.setJugadorQueUsa(seleccionado);
         menuJugador.getCrearPenca().setVisible(true);
         menuJugador.getCrearPenca().setEnabled(true);
         menuJugador.getPronosticar().setEnabled(true);
         menuJugador.getPronosticar().setVisible(true);
         menuJugador.getRankingss().setEnabled(true);
         menuJugador.getRankingss().setVisible(true);
         if(getSistema().isExistePenca()){
           menuJugador.getCrearPenca().setEnabled(false);
       }
         this.getSistema().setJugadorQueUsa(seleccionado);
         JOptionPane.showMessageDialog(this,"Usted se identifico como "+seleccionado.getAlias(),"Identificado",1);
         
         
     }else{
         JOptionPane.showMessageDialog(this,"Debe seleccionar un jugador","Atencion",2);
     }

    }//GEN-LAST:event_jButton1ActionPerformed
