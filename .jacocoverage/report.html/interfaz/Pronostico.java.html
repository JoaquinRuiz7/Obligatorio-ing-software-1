    private void finActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_finActionPerformed
        if(getSistema().seJugaron(getSistema().getPartidosSemis())){
             PronosticaFinal fin = new PronosticaFinal(this.getMenu(), true, getSistema());
        fin.setVisible(true);
        }else{
            JOptionPane.showMessageDialog(this, "Aun no se puede pronosticar en esta etapa , los partidos de semifinal no han finalizado.", "Alerta", 2);
        }
       
    }//GEN-LAST:event_finActionPerformed

    private void octavosActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_octavosActionPerformed

    }//GEN-LAST:event_octavosActionPerformed
