    private void jRegistrarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jRegistrarActionPerformed
        String nombre = this.jNombre.getText();
        if (Validaciones.validarAlias(nombre)) {
            Jugador nuevoJugador = new Jugador(nombre);
            if(sistema.registrarJugador(nuevoJugador)){
                JOptionPane.showMessageDialog(this,nuevoJugador.getAlias()+" has sido registrado con exito","Exito", 1);
                sistema.getJuego().agregarParticipante(nuevoJugador);
                sistema.guardarSistema(sistema);
            }else{
                JOptionPane.showMessageDialog(this,"El alias "+nuevoJugador.getAlias()+" ya esta en uso , elige otro.","No es posible registrase", 2);
            }
        }
    }//GEN-LAST:event_jRegistrarActionPerformed
