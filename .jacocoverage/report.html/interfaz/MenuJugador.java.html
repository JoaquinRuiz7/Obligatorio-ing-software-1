    private void rankingssActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_rankingssActionPerformed
        if(this.getSistema().getJuego()!=null){
            ArrayList<Partido> jugados = this.getSistema().resultadosEnfase();
            this.getSistema().sumarPuntajeFaseGrupos(jugados);
            getSistema().setPartidos(jugados);
            jugados = this.getSistema().partidosJugadosEnFases(16);
            this.getSistema().sumarPuntajeOctavos(jugados);
           getSistema().actualizar(getSistema().getPartidos8tavos());
            jugados = this.getSistema().partidosJugadosEnFases(8);
            this.getSistema().sumarPuntajeCuartos(jugados);
            getSistema().actualizar(getSistema().getPartidos4tos());
            jugados = this.getSistema().partidosJugadosEnFases(4);
            this.getSistema().sumarPuntajeCuartos(jugados);
            getSistema().actualizar(getSistema().getPartidosSemis());
            jugados = this.getSistema().finaly34Puesto();
            this.getSistema().sumarPuntajeCuartos(jugados);
            Collections.sort(this.getSistema().getJugadores());
            ArrayList<String> rankings = new ArrayList<String>();
            rankings = this.getSistema().rankingsumerados(this.getSistema().getJugadores());
            Rankings rankiing = new Rankings(this,true,getSistema(),rankings);
            rankiing.setVisible(true);
        }else{
            JOptionPane.showMessageDialog(this,"Debe parametrizar las reglas y crear la penca para poder acceder al ranking","Atencion",2);
        }

    }//GEN-LAST:event_rankingssActionPerformed
