package dominio;

import java.io.Serializable;


public class Reglas implements Serializable{
    private int aciertoGanador;
    private int aciertoEmpate;
    private int aciertoExacto;
    private boolean aumentoEnCambioDeFase;

    public int getAciertoGanador() {
        return aciertoGanador;
    }

    public void setAciertoGanador(int aciertoGanador) {
        this.aciertoGanador = aciertoGanador;
    }

    public int getAciertoExacto() {
        return aciertoExacto;
    }

    public void setAciertoExacto(int aciertoExacto) {
        this.aciertoExacto = aciertoExacto;
    }

    public boolean isAumentoEnCambioDeFase() {
        return aumentoEnCambioDeFase;
    }

    public void setAumentoEnCambioDeFase(boolean aumentoEnCambioDeFase) {
        this.aumentoEnCambioDeFase = aumentoEnCambioDeFase;
    }

    public int getAciertoEmpate() {
        return aciertoEmpate;
    }

    public void setAciertoEmpate(int aciertoEmpate) {
        this.aciertoEmpate = aciertoEmpate;
    }

    public Reglas(int aciertoGanador, int ae, int empate) {
        this.setAciertoGanador(aciertoGanador);
        this.setAciertoExacto(ae); 
        this.setAciertoEmpate(empate);
        this.setAumentoEnCambioDeFase(false);
    }
    public Reglas(){
        this.setAciertoGanador(0);
        this.setAciertoExacto(0);
        this.setAciertoEmpate(0);
        this.setAumentoEnCambioDeFase(false);
    }
    public void sumarAciertoExacto(Jugador jugador){
        jugador.setPuntaje(jugador.getPuntaje()+this.getAciertoExacto());
    }
    public void sumarAciertoEmpate(Jugador jugador){
        jugador.setPuntaje(jugador.getPuntaje()+this.getAciertoEmpate());
    }
    public void sumarAciertoGanador(Jugador jugador){
        jugador.setPuntaje(jugador.getPuntaje()+this.getAciertoGanador());
    }
   
  
}
