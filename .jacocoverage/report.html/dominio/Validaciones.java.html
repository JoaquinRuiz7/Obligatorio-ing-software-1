package dominio;

public class Validaciones {

    

    public static boolean letras(String texto) {
        texto.trim();
        return texto.matches("[A-Za-z ]*");
    }
    public static boolean validarAlias(String texto){
        texto.trim();
        return texto.matches("[A-Za-z0-9 ]*");
    }
    public static boolean validarCi(String ci) {
        if ((ci.length() > 8) || (ci.length() < 8)
                || (!ci.matches("[0-9]*"))) {
            return false;
        }
        ci.trim();
        return true;
    }

    public static boolean validarNumeros(String numeros) {
        return numeros.matches("[0-9]*");
    }

    public static boolean validarMaximoPuntaje(int numero) {
        return ((numero <= 10) && (numero > 0));
    }
}
