package dominio;

import java.io.Serializable;
import java.util.ArrayList;

public class Jugador implements Serializable, Comparable<Jugador> {

    private String alias;
    private int puntaje;
    private ArrayList<PronosticoPartido> pronosticos;
   
    public String getAlias() {
        return alias;
    }

    public void setAlias(String alias) {
        this.alias = alias;
    }

    public ArrayList<PronosticoPartido> getPronosticos() {
        return pronosticos;
    }

    public void setPronosticos(ArrayList<PronosticoPartido> pronosticos) {
        this.pronosticos = pronosticos;
    }

    public int getPuntaje() {
        return puntaje;
    }

    public void setPuntaje(int puntaje) {
        this.puntaje = puntaje;
    }

    public Jugador() {
        this.setAlias("");
        this.setPronosticos(new ArrayList<PronosticoPartido>());
        this.setPuntaje(0);
    }

    public Jugador(String alias) {
        this.setAlias(alias);
        this.setPuntaje(0);
        this.setPronosticos(new ArrayList<PronosticoPartido>());

    }

    @Override
    public String toString() {
        return this.getAlias();
    }

    @Override
    public boolean equals(Object o) {
        Jugador compara = (Jugador) o;
        return this.getAlias().equals(compara.getAlias());
    }

    @Override
    public int compareTo(Jugador jugador) {
        int diferencia = 0;
        diferencia = jugador.getPuntaje() - this.getPuntaje();
        return diferencia;
    }
}
