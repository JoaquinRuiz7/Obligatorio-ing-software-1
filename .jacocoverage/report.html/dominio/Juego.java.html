package dominio;

import java.io.Serializable;
import java.util.ArrayList;

public class Juego implements Serializable{
    private ArrayList<Jugador> participantes;
    private Reglas reglas;
    private String nombreDePenca;
    public ArrayList<Jugador> getParticipantes() {
        return participantes;
    }

    public void setParticipantes(ArrayList<Jugador> participantes) {
        this.participantes = participantes;
    }

    public Reglas getReglas() {
        return reglas;
    }

    public String getNombreDePenca() {
        return nombreDePenca;
    }

    public void setNombreDePenca(String nombreDePenca) {
        this.nombreDePenca = nombreDePenca;
    }

    public void setReglas(Reglas reglas) {
        this.reglas = reglas;
    }

   

  
    
    public Juego(ArrayList<Jugador> participantes, Reglas reglas,String nombre ) {
        this.setParticipantes(participantes);
        this.setReglas(reglas);
        this.setNombreDePenca(nombre);
       
        
    }

    public Juego() {
        this.setParticipantes(new ArrayList<Jugador>());
        this.setReglas(new Reglas());
        this.setNombreDePenca("");
        
    }

    public void agregarParticipante(Jugador participante) {
        if (!(this.getParticipantes().contains(participante))) {
                this.getParticipantes().add(participante);
        }
    }
    @Override
    public String toString(){
        return this.getNombreDePenca();
    }
}
