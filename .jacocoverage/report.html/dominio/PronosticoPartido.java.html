public class PronosticoPartido implements Serializable {

    public int[] getResultado() {
        return resultado;
    }

   
    public Partido getPartido() {
        return partido;
    }

    public void setPartido(Partido partido) {
        this.partido = partido;
    }

   

    public void setResultado(int[] resultado) {
        this.resultado = resultado;
    }

   

    public void setEquipoGanador(String equipoGanador) {
        equipoGanador = equipoGanador;
    }

    public PronosticoPartido(int[] resultado, Partido partido) {
        this.setResultado(resultado);
        this.setPartido(partido);   
    }

   

    public PronosticoPartido() {
        this.setResultado(new int[2]);
        this.setPartido(new Partido());
    }
    @Override
    public boolean equals(Object o) {
        PronosticoPartido compara = (PronosticoPartido) o;
        return this.getPartido().equals(compara.getPartido());
    }
    @Override
    public String toString(){
        return this.getPartido().getEquipos()[0]+" "+this.getResultado()[0]+
                " VS "+this.getPartido().getEquipos()[1]+" "+this.getResultado()[1];
    }
    
}
